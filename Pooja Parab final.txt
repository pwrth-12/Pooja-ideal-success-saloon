let selectedRating = 0;
const ratingStars = document.querySelectorAll('#reviews .rating-star');
const submitBtn = document.getElementById('submitRatingBtn');
const messageDiv = document.getElementById('ratingMessage');

const hasRated = localStorage.getItem('hasRatedIdealSuccess');
if (hasRated === 'true') {
  ratingStars.forEach(star => {
    star.classList.add('disabled');
    star.style.pointerEvents = 'none';
  });
  submitBtn.disabled = true;
  messageDiv.innerText = 'You have already rated us. Thank you for your feedback! 🙏';
}

if (hasRated !== 'true') {
  ratingStars.forEach(star => {
    star.addEventListener('click', () => {
      const val = parseInt(star.getAttribute('data-value'));
      selectedRating = val;
      ratingStars.forEach((s, idx) => {
        if (idx < val) {
          s.classList.add('selected');
          s.innerText = '★';
        } else {
          s.classList.remove('selected');
          s.innerText = '☆';
        }
      });
    });
  });
}

submitBtn.addEventListener('click', () => {
  if (hasRated === 'true') return;
  if (selectedRating === 0) {
    messageDiv.innerText = 'Please select a star rating first.';
    setTimeout(() => { messageDiv.innerText = ''; }, 2000);
    return;
  }
  localStorage.setItem('userRatingIdealSuccess', selectedRating);
  localStorage.setItem('hasRatedIdealSuccess', 'true');
  messageDiv.innerText = 'Thank you for your rating! 🙏';
  ratingStars.forEach(star => {
    star.classList.add('disabled');
    star.style.pointerEvents = 'none';
  });
  submitBtn.disabled = true;
});
